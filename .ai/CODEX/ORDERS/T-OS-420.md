### 2. `.claude/hooks/pretool_policy.py` 拡張

```python
def get_invariant_mode(invariant_id: str, config: dict) -> str:
    """Per-invariant mode lookup with default fallback."""
    invariants = config.get("invariants", {})
    return invariants.get(invariant_id, config.get("default", "warn"))

# enforcement で:
for invariant_id, reason in violations:
    mode = get_invariant_mode(invariant_id, config)
    if mode == "disabled":
        continue
    elif mode == "warn":
        print(f"ORGOS_POLICY_WARN: {invariant_id}: {reason}", file=sys.stderr)
    elif mode == "enforce":
        print(f"ORGOS_POLICY_DENY: {invariant_id}: {reason}", file=sys.stderr)
        return 2
```
