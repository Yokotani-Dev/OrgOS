```python
def evaluate_invariant_violations(tool, command, path, cwd):
    """
    Returns list of (invariant_id, reason) tuples.
    Empty list = no violation.
    """
    violations = []
    
    if tool == "Bash":
        git = parse_git_command(command)
        if git:
            # Invariant #1: Integrator-Only Commit
            if git.subcmd in {"commit", "push"}:
                violations.append(("IntegratorOnlyCommit", f"raw git {git.subcmd} is blocked"))
            
            # Invariant #3: Protected Branch No-Touch
            if git.subcmd == "reset" and "--hard" in git.args:
                violations.append(("ProtectedBranchNoTouch", "git reset --hard is blocked"))
            
            if git.subcmd == "branch" and any(a in git.args for a in ["-f", "-D"]):
                violations.append(("ProtectedBranchNoTouch", "branch mutation is blocked"))
            
            if git.subcmd in {"checkout", "switch"}:
                target = infer_git_target(git.args)
                if target in {"main", "develop"}:
                    violations.append(("ProtectedBranchNoTouch", f"checkout/switch to {target} blocked"))
            
            if git.subcmd in {"merge", "rebase", "cherry-pick"}:
                violations.append(("IntegratorOnlyCommit", f"git {git.subcmd} requires integrator script"))
            
            if git.subcmd == "worktree" and any(a in git.args for a in ["add", "remove", "prune"]):
                violations.append(("PerTaskWorktree", "worktree mutation requires org wrapper"))
        
        # Dangerous shell patterns
        if re.search(r'\brm\s+-rf\s+/', command) or re.search(r'curl[^|]+\|\s*(ba)?sh', command):

    elif tool in {"Edit", "Write"}:
        # Invariant #5: State Mutation via Org Tool
        if is_protected_state_file(path):
            violations.append(("StateMutationViaOrgTool", f"direct edit of {path} is blocked"))
    
    return violations
```

#### 1-3. mode-based enforcement

```python
def enforce_violations(violations, mode):
    """
    mode=enforce: print + exit 2 on first violation
    mode=warn: print warning, allow execution
    mode=disabled: do nothing
    """
    if not violations or mode == "disabled":
        return 0
    
    inv_id, reason = violations[0]  # first violation
    
    if mode == "warn":
        print(f"ORGOS_POLICY_WARN: {inv_id}: {reason}", file=sys.stderr)
        print(f"  (kernel mode={mode}; would be blocked in enforce mode)", file=sys.stderr)
        return 0
    
    if mode == "enforce":
        print(f"ORGOS_POLICY_DENY: {inv_id}: {reason}", file=sys.stderr)
        return 2
    
    return 0
```

## Important: Self-blocking 回避

Codex 自身が本 task で `pretool_policy.py` を書き換えるが、**書き換え後の policy は Codex の Edit/Write には適用されない**:
- Codex は worktree 内で実装している
- worktree 内の `.claude/hooks/pretool_policy.py` を Edit/Write する
- pretool hook は **main repo 側** の policy ファイルを読む (Codex の Claude Code セッションが main repo 側で起動されているため)
- したがって Codex 自身が自分の Edit を block することはない

ただし、本 task 完了後 Manager が cp で main repo に反映する瞬間、main の policy が新版になる。**Manager の次の Edit/Write はその新 policy で評価される** が、`mode=warn` default なので警告のみ。Owner が `enforce` に flip するまでは Manager commit も通る。
