def main():
    # Parse input
    if "--test-fixture" in sys.argv:
        idx = sys.argv.index("--test-fixture")
        with open(sys.argv[idx + 1]) as f:
            tool_input = json.load(f)
    else:
        tool_input = json.load(sys.stdin)
    
    tool = tool_input.get("tool", "")
    command = tool_input.get("command", "")
    path = tool_input.get("path", "")
    cwd = tool_input.get("cwd", os.getcwd())
    
    # Gather context
    actor = get_actor_role(tool_input)
    leases = load_active_leases(cwd)
    config = load_mode_config()
    
    # Evaluate
    decision = policy_core.evaluate(tool, command, path, cwd, actor, leases, config)
    
    # Output
    if decision.outcome == "allow":
        return 0
    elif decision.outcome == "warn":
        print(f"ORGOS_POLICY_WARN: {decision.invariant_id}: {decision.reason}", file=sys.stderr)
        return 0
    else:  # deny
        print(f"ORGOS_POLICY_DENY: {decision.invariant_id}: {decision.reason}", file=sys.stderr)
        return 2


if __name__ == "__main__":
    sys.exit(main())
